下面的示例用一个轻量检查清单描述如何把「隐私合规与数据治理」落到工程评审中：

```ts
interface ArticleCheckpoint {
  /** 检查项名称。 */
  title: string
  /** 检查项需要满足的要求。 */
  requirement: string
  /** 当前检查项是否已经完成。 */
  done: boolean
}

/**
 * 筛选仍未完成的检查项。
 * @param checkpoints 当前主题的工程检查清单。
 * @returns 尚未完成的检查项列表。
 */
function getUnfinishedCheckpoints(checkpoints: ArticleCheckpoint[]): ArticleCheckpoint[] {
  return checkpoints.filter((checkpoint) => {
    // 中文注释：只保留未完成项，便于在评审或上线前集中处理风险。
    return !checkpoint.done
  })
}

const checkpoints: ArticleCheckpoint[] = [
  {
    title: "概念边界",
    requirement: "已经说明适用场景、限制条件和替代方案。",
    done: true,
  },
  {
    title: "工程验证",
    requirement: "已经准备测试、日志或性能指标来验证结论。",
    done: false,
  },
]

console.table(getUnfinishedCheckpoints(checkpoints))
```

| 误区 | 更可靠的处理方式 |
