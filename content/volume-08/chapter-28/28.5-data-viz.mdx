interface ArticleCheckpoint {
  /** 检查项名称。 */
  title: string
  /** 检查项需要满足的要求。 */
  requirement: string
  /** 当前检查项是否已经完成。 */
  done: boolean
}

/**
 * 筛选仍未完成的检查项。
 * @param checkpoints 当前主题的工程检查清单。
 * @returns 尚未完成的检查项列表。
 */
function getUnfinishedCheckpoints(checkpoints: ArticleCheckpoint[]): ArticleCheckpoint[] {
  return checkpoints.filter((checkpoint) => {
    // 中文注释：只保留未完成项，便于在评审或上线前集中处理风险。
    return !checkpoint.done
  })
}

## 常见误区

| 误区 | 更可靠的处理方式 |
|---|---|
| 只看视觉效果而不测低端设备，会把渲染成本留给用户。 | 把问题拆到上下文、边界、证据和回退方案中验证。 |
| 大量 DOM 渲染海量数据 | 先用最小场景复现，再决定是否升级方案。 |
| 忽略高 DPR 内存占用 | 先用最小场景复现，再决定是否升级方案。 |
